    {
        policy
            .WithOrigins("http://localhost:5173") // tu Vite
            .AllowAnyHeader()
            .AllowAnyMethod();

{
    app.MapOpenApi();
    app.UseSwaggerUI();
    app.UseSwagger();
}

// Usar la pol√≠tica de CORS (MUY IMPORTANTE: antes de UseAuthorization)
